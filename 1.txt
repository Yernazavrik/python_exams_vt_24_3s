import math
def st(base, index):
	if  index == 0:
		return 1
	return base * st(base, index-1)
def fact(i):
	if  i == 0:
		return 1
	return i * fact(i-1)
def rec(x, n):
	if n == 0:
		return 0
	return (st(x, n) / fact(n*n)) + rec(x, n-1)
a = int(input())
b = int(input())
print(rec(a,b))
